<chunk> ::= {<stat>} [<laststat>]
<stat> ::= <whilestat> | <ifstat> | <forstat> | <functionstat> | <vardefstat> | <varassignstat> | <wrap> | <StatementSeperator>

<whilestat> ::= <While> <equation> <wrap>
<ifstat> ::= <If> <equation> <wrap> {<ElseIf> <equation> <wrap>} [<Else> <wrap>]
<forstat> ::= <For> <Symbol> <forbody> <wrap>
<functionstat> ::= [<NoRefresh>] <Define> <function>
<vardefstat> ::= [<Shared>] <Type> <Symbol> <StatementSeperator>
<varassignstat> ::= <var> <assignorcall>

<assignorcall> ::= <Assign> <equation> | <functioncall>
<forbody> ::= "=" <equation> <FieldSeperator> <equation> <FieldSeperator> <equation> | <In> <var>
<functioncall> ::= <args> {<args>}
<namelist> ::= <Symbol> {<FieldSeperator> <Symbol>}
<tableconstructor> ::= "[" [<explist1>] "]"
<wrap> ::= "{" [<chunk>] "}"
<function> ::= <Symbol> <funcbody>
<funcbody> ::= "(" [<paramlist>] ")" <wrap>
<slice> ::= "[" <equation> "]"
// <equation> ::= <bracketequation> {<Binop> <equation>} | [<Unop>] <literals> {<Binop> <equation>}
// <bracketequation> ::= "(" [<Unop>] <literals> {<Binop> <equation>} ")"
<equation> ::= <comparison>
<comparison> ::= <addition> { ("=" | ">" | "<") <addition> }
<addition> ::= <multiplication> { ("+" | "-") <multiplication> }
<multiplication> ::= <unary> { ("*" | "/") <unary> }
<unary> ::= [<Unop>] <primary>
<primary> ::= <literal> | "(" <equation> ")"
<var> ::= <Symbol> {<Dot> <Symbol>} [<slice>]
<literals> ::= <Bool> | <Number> | <String> | <var> [<functioncall>] | <tableconstructor>
<explist1> ::= <equation> {<FieldSeperator> <explist1>}
<laststat> ::= <Break> | <StatementSeperator> | <Return> [<explist1>] |
<varlist1> ::= <literals> {<FieldSeperator> <literals>}
<args> ::= "(" [<varlist1>] ")"
<argtype> ::= <Symbol> <Colon> <Type>
<paramlist> ::= <argtype> {<FieldSeperator> <argtype>}
