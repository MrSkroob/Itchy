shared number score
shared number level
list levels
number speed


event event_whenflagclicked() {
    score = 0
    level = 1
    speed = 10

    data_deletealloflist(levels)

    data_addtolist(10, levels)
    data_addtolist(12, levels)
    data_addtolist(14, levels)
    data_addtolist(16, levels)

    data_showvariable(score)
    data_showvariable(level)
    speed = levels[level]
    motion_gotoxy(0, 0)

    motion_pointindirection(operator_random(-180, 180))

    while true {
        motion_movesteps(speed)

        if motion_yposition() < -150 {
            control_stop("all")
        }
    }
}

// collision detection
event event_whenflagclicked() {
    while true {
        motion_ifonedgebounce()

        if sensing_touchingobject("Paddle") {
            score += 1

            if score > 10 {
                score = 0
                level += 1
                if level > data_lengthoflist(levels) {
                    speed = levels[data_lengthoflist(levels)]
                }
                else {
                    speed = levels[level]
                }
            }

            motion_pointindirection(180 - motion_direction())
            control_wait_until(not sensing_touchingobject("Paddle"))
        }
    }
}
